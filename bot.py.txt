def load_data():
    if os.path.exists(DATA_FILE):
        with open(DATA_FILE, "r", encoding="utf-8") as f:
            return json.load(f)
    return {}

def save_data(data):
    with open(DATA_FILE, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)

def get_user_key(user_id):
    return f"user_{user_id}"

def is_current_month(d: str) -> bool:
    """–ü—Ä–æ–≤–µ—Ä—è–µ—Ç, –æ—Ç–Ω–æ—Å–∏—Ç—Å—è –ª–∏ –¥–∞—Ç–∞ (–≤ —Ñ–æ—Ä–º–∞—Ç–µ 'YYYY-MM-DD') –∫ —Ç–µ–∫—É—â–µ–º—É –º–µ—Å—è—Ü—É."""
    try:
        plan_date = datetime.strptime(d, "%Y-%m-%d").date()
        today = date.today()
        return plan_date.year == today.year and plan_date.month == today.month
    except:
        return False

    """–≠–∫—Å–ø–æ—Ä—Ç–∏—Ä—É–µ—Ç –≤—ã–ø–æ–ª–Ω–µ–Ω–Ω—ã–µ –ø–ª–∞–Ω—ã –∑–∞ —É–∫–∞–∑–∞–Ω–Ω—ã–π –º–µ—Å—è—Ü –≤ Excel."""
    data = load_data()
    filename = REPORTS_DIR / f"{year}-{month:02d}.xlsx"
    
    wb = openpyxl.Workbook()
    ws = wb.active
    ws.title = f"{year}-{month:02d}"
    ws.append(["–î–∞—Ç–∞ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è", "–ü–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å ID", "–ü–ª–∞–Ω"])

def cleanup_old_data():
    """–£–¥–∞–ª—è–µ—Ç –≤—Å–µ –∑–∞–ø–∏—Å–∏, –∫—Ä–æ–º–µ —Ç–µ—Ö, —á—Ç–æ –æ—Ç–Ω–æ—Å—è—Ç—Å—è –∫ —Ç–µ–∫—É—â–µ–º—É –º–µ—Å—è—Ü—É."""
    data = load_data()
    today = date.today()
    current_year, current_month = today.year, today.month

    for user_key in list(data.keys()):
        user_data = data[user_key]

        # –û—Å—Ç–∞–≤–ª—è–µ–º —Ç–æ–ª—å–∫–æ –∞–∫—Ç–∏–≤–Ω—ã–µ –ø–ª–∞–Ω—ã —Ç–µ–∫—É—â–µ–≥–æ –º–µ—Å—è—Ü–∞
        if "active_plans" in user_data:
            user_data["active_plans"] = [
                p for p in user_data["active_plans"]
                if is_current_month(p.get("date_added", ""))
            ]

        # –û—Å—Ç–∞–≤–ª—è–µ–º —Ç–æ–ª—å–∫–æ –≤—ã–ø–æ–ª–Ω–µ–Ω–Ω—ã–µ –ø–ª–∞–Ω—ã —Ç–µ–∫—É—â–µ–≥–æ –º–µ—Å—è—Ü–∞
        if "completed_plans" in user_data:
            user_data["completed_plans"] = [
                p for p in user_data["completed_plans"]
                if is_current_month(p.get("date_completed", ""))
            ]

        # –£–¥–∞–ª—è–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è, –µ—Å–ª–∏ —É –Ω–µ–≥–æ –Ω–∏—á–µ–≥–æ –Ω–µ –æ—Å—Ç–∞–ª–æ—Å—å
        if not user_data.get("active_plans") and not user_data.get("completed_plans"):
            del data[user_key]

    save_data(data)

@router.message()
async def add_plan(message: Message):
    user_id = message.from_user.id
    text = message.text.strip()
    if not text:
        return

    data = load_data()
    user_key = get_user_key(user_id)

    if user_key not in data:
        data[user_key] = {"active_plans": [], "completed_plans": []}

    new_plan = {
        "text": text,
        "date_added": str(date.today())
    }
    data[user_key]["active_plans"].append(new_plan)
    save_data(data)

    kb = InlineKeyboardMarkup(inline_keyboard=[
        [
            InlineKeyboardButton(text="‚úÖ –í—ã–ø–æ–ª–Ω–µ–Ω–æ", callback_data=f"done_{len(data[user_key]['active_plans'])-1}"),
            InlineKeyboardButton(text="‚ùå –£–¥–∞–ª–∏—Ç—å", callback_data=f"del_{len(data[user_key]['active_plans'])-1}")
        ]
    ])

    await message.answer(f"üìù –ü–ª–∞–Ω –¥–æ–±–∞–≤–ª–µ–Ω:\n<b>{text}</b>", parse_mode=ParseMode.HTML, reply_markup=kb)

@router.callback_query(lambda c: c.data.startswith("done_") or c.data.startswith("del_"))
async def handle_action(callback: CallbackQuery):
    user_id = callback.from_user.id
    user_key = get_user_key(user_id)
    data = load_data()

    if user_key not in data:
        await callback.answer("–ù–µ—Ç –∞–∫—Ç–∏–≤–Ω—ã—Ö –ø–ª–∞–Ω–æ–≤.", show_alert=True)
        return

    action, index_str = callback.data.split("_", 1)
    try:
        index = int(index_str)
        active_plans = data[user_key]["active_plans"]
        if index < 0 or index >= len(active_plans):
            raise IndexError
        plan = active_plans.pop(index)
    except (ValueError, IndexError):
        await callback.answer("–ü–ª–∞–Ω —É—Å—Ç–∞—Ä–µ–ª –∏–ª–∏ —É–∂–µ —É–¥–∞–ª—ë–Ω.", show_alert=True)
        await callback.message.delete()
        return

    if action == "done":
        plan["date_completed"] = str(date.today())
        data[user_key].setdefault("completed_plans", []).append(plan)
        await callback.message.edit_text("‚úÖ –í—ã–ø–æ–ª–Ω–µ–Ω–æ! –°–æ—Ö—Ä–∞–Ω–µ–Ω–æ –≤ –æ—Ç—á—ë—Ç.")

    elif action == "del":
        await callback.message.edit_text("‚ùå –£–¥–∞–ª–µ–Ω–æ.")

    save_data(data)
    await callback.answer()

        prev_year = today.year if today.month > 1 else today.year - 1
        report_file = export_month_to_excel(prev_year, prev_month)

    else:

# ============ WEBHOOK –ò –ó–ê–ü–£–°–ö ============
async def on_startup(app):
    await bot.set_webhook(WEBHOOK_URL)

def main():
    dp.include_router(router)
    app = web.Application()
    webhook_requests_handler = SimpleRequestHandler(dispatcher=dp, bot=bot)
    webhook_requests_handler.register(app, path=WEBHOOK_PATH)
    app.router.add_get("/trigger-cleanup", trigger_cleanup)
    app.on_startup.append(on_startup)
    web.run_app(app, host="0.0.0.0", port=PORT)

if __name__ == "__main__":
    main()
